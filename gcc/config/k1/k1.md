(define_expand "store_multiple"
  [(match_par_dup 3 [(set (match_operand:DI 0 "" "")
			  (match_operand:DI 1 "" ""))
		     (use (match_operand:DI 2 "" ""))])]
  ""
  "
{
  int regno;
  int count;
  rtx to;
  rtx op0;
  int i;

  count = INTVAL (operands[2]);
  regno = REGNO (operands[1]);

  if (GET_CODE (operands[2]) != CONST_INT
      || GET_MODE(operands[1]) != DImode
      || (count != 2 && !(count == 4))
      || ((count == 2) && (regno & 1))
      || ((count == 4) && (regno & 3))
      || !MEM_P (operands[0])
      || !REG_P (operands[1])
      || MEM_ALIGN (operands[0]) < (count * UNITS_PER_WORD)
      || MEM_VOLATILE_P (operands[0])
      || REGNO (operands[1]) > K1C_GPR_LAST_REGNO)
    FAIL;


  operands[3] = gen_rtx_PARALLEL (VOIDmode, rtvec_alloc (count /* + 1 */));
  to = force_reg (DImode, XEXP (operands[0], 0));
  op0 = replace_equiv_address (operands[0], to);

  XVECEXP (operands[3], 0, 0)
    = gen_rtx_SET (adjust_address_nv (op0, DImode, 0), operands[1]);

/*  XVECEXP (operands[3], 0, 1) = gen_rtx_CLOBBER (VOIDmode,
 *						 gen_rtx_SCRATCH (DImode));
 */
  for (i = 1; i < count; i++)
    XVECEXP (operands[3], 0, i /* + 1 */)
      = gen_rtx_SET (adjust_address_nv (op0, DImode, i * UNITS_PER_WORD),
		     gen_rtx_REG (DImode, regno + i));
}")


(define_expand "load_multiple"
  [(match_par_dup 3 [(set (match_operand:DI 0 "" "")
			  (match_operand:DI 1 "" ""))
		     (use (match_operand:DI 2 "" ""))])]
  ""
  "
{
  int regno;
  int count;
  rtx op1;
  int i;

  count = INTVAL (operands[2]);
  regno = REGNO (operands[0]);

  /* Only supports loads of 2 or 4 registers, correctly aligned */
  if (GET_CODE (operands[2]) != CONST_INT
      || GET_MODE(operands[0]) != DImode
      || (count != 2 && !(count == 4))
      || ((count == 2) && (regno & 1))
      || ((count == 4) && (regno & 3))
      || !MEM_P (operands[1])
      || !REG_P (operands[0])
      || MEM_ALIGN (operands[1]) < (count * UNITS_PER_WORD)
      || MEM_VOLATILE_P (operands[1])
      || REGNO (operands[0]) > K1C_GPR_LAST_REGNO)
    FAIL;

  operands[3] = gen_rtx_PARALLEL (VOIDmode, rtvec_alloc (count));

  rtx base, offset;
  if (! k1_split_mem (XEXP (operands[1], 0), &base, &offset, false))
    FAIL;

  if (! REG_P (base))
    base = force_reg (Pmode, base);

  /* Add a PLUS so that we have a simpler match in load multiple patterns */
  XEXP(operands[1], 0) = gen_rtx_PLUS (Pmode, base, offset);

  for (i = 0; i < count; i++)
    XVECEXP (operands[3], 0, i)
      = gen_rtx_SET (gen_rtx_REG (DImode, regno + i),
		     adjust_address_nv (operands[1], DImode, i * UNITS_PER_WORD));
}")

	} else if (GET_CODE (operands[1]) == UNLE) {
	   PUT_CODE (operands[1], UNGE);
	   swap = 1;
	} else if (GET_CODE (operands[1]) == UNGT) {
	   PUT_CODE (operands[1], UNLT);
	   swap = 1;
	} 

	if (swap) {
	   emit_insn (gen_fcomp<ALLF:sfx> (operands[0], operands[1],
	   	     			       operands[3], operands[2]));
	   DONE;
	}

	if (GET_CODE (operands[1]) == UNORDERED) {
	   rtx tmp = gen_reg_rtx (DImode);
	   rtx comp2 = copy_rtx (operands[1]);
	   PUT_CODE (operands[1], UNGE);
	   emit_insn (gen_fcomp<ALLF:sfx> (operands[0], operands[1],
	   	     			       operands[2], operands[3]));
	   PUT_CODE (comp2, UNLT);
	   emit_insn (gen_fcomp<ALLF:sfx> (tmp, comp2,
	   	     			       operands[2], operands[3]));

(define_expand "doloop_end"
  [(parallel [(set (pc) (if_then_else
                          (ne (match_operand:SI 0 "" "")
                              (const_int 1))
                          (label_ref (match_operand 1 "" ""))
                          (pc)))
              (set (match_dup 0)
                   (plus:SI (match_dup 0)
                            (const_int -1)))
              (unspec [(const_int 0)] UNSPEC_LOOPDO_END)
              (clobber (match_scratch:SI 2))])]
  ""
{
  /* The loop optimizer doesn't check the predicates... */
  if (GET_MODE (operands[0]) != SImode)
    FAIL;
  operands[2] = gen_rtx_SCRATCH (SImode);
})
